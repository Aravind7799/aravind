<script>
function create(){
  if(!from.value || !to.value){
    alert("Please fill both names");
    return;
  }
  let link = location.origin + location.pathname +
    `?f=${encodeURIComponent(from.value)}&t=${encodeURIComponent(to.value)}`;
  show(from.value,to.value,link);
}

function show(f,t,l){
  form.style.display="none";
  card.style.display="block";
  toText.innerText="Dear " + t + ",";
  fromText.innerText=f;
  link.innerText=l;
}

function copy(){
  navigator.clipboard.writeText(link.innerText);
  alert("Link copied ‚ù§Ô∏è");
}
